# Raspberry Pi 4 Model B
#
# Boot flow with TF-A:
#   GPU firmware loads bl31.bin (TF-A BL31) at EL3
#   TF-A patches DTB to advertise PSCI (method=smc)
#   TF-A drops Core 0 into EL2 at 0x80000 (our hypervisor)
#   TF-A parks Cores 1-3 in its own secondary spin loop
#   Linux calls CPU_ON SMC -> TF-A wakes Cores 1-3 at EL1
#   Result: all 4 cores enter Linux at EL1 (no mode mismatch)
#
armstub=bl31.bin
arm_64bit=1
kernel=kernel8.img
enable_uart=1
dtoverlay=disable-bt
hdmi_force_hotplug=1
hdmi_drive=2
disable_overscan=1
hdmi_group=1
hdmi_mode=4
framebuffer_width=1280
framebuffer_height=720
framebuffer_depth=32
disable_splash=1
